方法一：NewMark

function Res = Newmark(Acc,DT,T,kesi,IND)
    % parameters
    NT = max(size(T,1),size(T,2));
    NPTS = max(size(Acc,1),size(Acc,2));
    DDY = Acc;
    %% Maxima of input motion
    Emax(1) =max( abs(DDY));
    Emax(2) = 0.0;
    Emax(3) = 0.0;
    DDYF = DDY(1);
    DYF = 0.0;
    YF = 0.0;
    %% Computation of the maximun of a,v,d of the ground motion
%     for i = 2:NPTS
%         DDYM = DDY(i);
%         DY = DYF + (DDYF+DDYM)*DT/2.0;
%         Y = YF + DYF*DT + (DDYF/3.+ DDYM/6.)*DT*DT;
%         Emax(1) = max(Emax(1),abs(DDYM));
%         Emax(2) = max(Emax(2),abs(DY));
%         Emax(3) = max(Emax(3),abs(Y));
%         DDYF  = DDYM;
%         DYF = DY;
%         YF = Y;
%     end
    %Qmax = Emax(IND);%
    %% Response Computation
    H =kesi;
    Res = zeros(1,NT);
    for i = 1:NT
        if (T(i) ==0.0 & IND == 1)
                Res(i) = Emax(1);
        elseif(T(i) ==0.0 &  IND ~= 1)
                Res(i) =0.0;
        else
           W = pi*2/T(i);
           W2 = W*W;
           HW = H*W;
           WD = W*sqrt(1-H.^2);
           WDT = WD*DT;
           E = exp(-HW*DT);
           CWDT = cos(WDT);
           SWDT = sin(WDT);
           A11 = E*(CWDT+HW*SWDT/WD);
           A12 = E*SWDT/WD;
           A21 = -E*W2*SWDT/WD;
           A22 = E*(CWDT-HW*SWDT/WD);
           SS = -HW*SWDT - WD*CWDT;
           CC = -HW*CWDT + WD*SWDT;
           S1 = (E*SS + WD)/W2;
           C1 = (E*CC + HW)/W2;
           S2 = (E*DT*SS + HW*S1 + WD*C1)/W2;
           C2 = (E*DT*CC + HW*C1 - WD*S1)/W2;
           S3 = DT*S1-S2;
           C3 = DT*C1-C2;
           B11 = -S2/WDT;
           B12 = -S3/WDT;
           B21 = (HW*S2 - WD*C2)/WDT;
           B22 = (HW*S3 - WD*C3)/WDT;
           Rmax(1) = 2.*HW*abs(DDY(1))*DT;
           Rmax(2) = abs(DDY(1))*DT;
           Rmax(3) = 0.0;
           DXF = -DDY(1)*DT;
           XF = 0.0;
           for k = 2:NPTS
               DDYM = DDY(k);
               DDYF = DDY(k-1);
               X = A12*DXF + A11*XF + B12*DDYM + B11*DDYF;
               DX = A22*DXF + A21*XF + B22*DDYM + B21*DDYF;
               DDX = -2.*HW*DX - W2*X;
               Rmax(1) = max(Rmax(1),abs(DDX));
               Rmax(2) = max(Rmax(2),abs(DX));
               Rmax(3) = max(Rmax(3),abs(X));
               DXF = DX;
               XF = X;
           end
           Res(i)=Rmax(IND);
        end
    end
end

 S1= [S1;S];
